bismarckevents.1 = {
	type = country_event
	placement = ROOT

	title = bismarckevents.1.t
	desc = bismarckevents.1.d
	flavor = bismarckevents.1.f

	gui_window = event_window_1char_tabloid

	left_icon = scope:bismarck_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	cooldown = { days = never_fire_again_modifier_time }

	duration = 3

	trigger = {
		# triggered by character spawn
	}

	immediate = {
		create_character = {
			template = PRU_otto_von_bismarck
			on_created = {
				set_variable = bismarck_variable_char
				set_global_variable = bismarck_variable
			}
		}
		random_scope_character = {
			limit = {
				has_variable = bismarck_variable_char
			}
			save_scope_as = bismarck_scope
		}
		ig:ig_landowners ?= {
			save_scope_as = junkers
		}
	}

	option = {
		name = bismarckevents.1.a
		default_option = yes
		scope:bismarck_scope = {
			add_modifier = {
				name = bismarck_popular
			}
			if = {
				limit = {
					has_variable = memento_mori_immunity
				}
				remove_variable = memento_mori_immunity
			}
			set_variable = { name = memento_mori_immunity years = 30 }
		}
		add_modifier = {
			name = realpolitik
			days = long_modifier_time
		}

		ai_chance = {
			base = 95
		}
	}
	option = {
		name = bismarckevents.1.b
		show_as_tooltip = {
			create_character = {
				template = PRU_otto_von_bismarck
			}
		}
		scope:bismarck_scope = {
			add_modifier = {
				name = bismarck_unpopular
			}
		}
		add_modifier = {
			name = germany_aggressive_expansion
			days = long_modifier_time
		}

		ai_chance = {
			base = 5
		}
	}
}

bismarckevents.7 = { #Rhineland general strikes strikes
	type = country_event
	placement = ROOT

	title = bismarckevents.7.t
	desc = bismarckevents.7.d
	flavor = bismarckevents.7.f

	cooldown = { days = stupidly_long_modifier_time }

	event_image = {
		video = "africa_soldiers_breaking_protest"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/alert_icons/radicalized.dds"

	duration = 3

	trigger = { 
		je:je_bismarck_reform ?= {
			"scripted_bar_progress(socialist_tension_bismarck_bar)" >= 50
		}
		any_state = {
			OR = {
				state_region = s:STATE_RHINELAND
				state_region = s:STATE_NORTH_RHINE
			}
		}
	}

	immediate = {
	}

	option = { #let them strike
		name = bismarckevents.7.a
		default_option = yes
		every_state = {
			limit = {
				owner = root
				OR = {
					state_region = s:STATE_RHINELAND
					state_region = s:STATE_NORTH_RHINE
				}
			}
			add_modifier = {
				name = strikes_bismarck
				months = 24
				is_decaying = yes
			}
		}
	}
	option = { #break the strikes
		name = bismarckevents.7.b
		trigger = {
			has_law_or_variant = law_type:law_national_guard
		}
		every_state = {
			limit = {
				owner = root
				OR = {
					state_region = s:STATE_RHINELAND
					state_region = s:STATE_NORTH_RHINE
				}
			}
			add_modifier = {
				name = strikes_bismarck
				months = 24
				multiplier = 0.5
				is_decaying = yes
			}
			add_radicals_in_state = {
				pop_type = laborers
				value = 0.15
			}
		}
	}
}
